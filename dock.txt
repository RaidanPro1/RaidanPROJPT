=== Dockerfile.frontend ===
# ðŸ‡¾ðŸ‡ª RaidanPro Frontend - Dockerfile
# Stage 1: Build
FROM node:20-alpine AS builder
WORKDIR /app
# Copy package manifests first for caching
COPY package.json package-lock.json* ./
RUN npm install
# Copy source
COPY . .
# Build the app (Vite)
RUN npm run build

# Stage 2: Serve
FROM nginx:alpine
# Copy build artifacts
COPY --from=builder /app/dist /usr/share/nginx/html
# SPA Routing Config
RUN echo 'server { \
    listen 80; \
    server_name localhost; \
    root /usr/share/nginx/html; \
    index index.html; \
    location / { \
        try_files $uri $uri/ /index.html; \
    } \
}' > /etc/nginx/conf.d/default.conf

EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]

=== backend/Dockerfile ===
# ðŸ‡¾ðŸ‡ª RaidanPro Backend & Orchestrator - Dockerfile
FROM python:3.11-slim

# Set environment variables
ENV PYTHONDONTWRITEBYTECODE=1 \
    PYTHONUNBUFFERED=1 \
    DEBIAN_FRONTEND=noninteractive

# Install System Dependencies
# docker.io: Required for the Installer Wizard to orchestrate the host via socket
# curl/git: For downloading models and repos
# ffmpeg: For audio processing (Whisper)
RUN apt-get update && apt-get install -y --no-install-recommends \
    curl \
    git \
    docker.io \
    openssl \
    ffmpeg \
    procps \
    && rm -rf /var/lib/apt/lists/*

WORKDIR /app

# Install Python Dependencies
COPY requirements.txt .
RUN pip install --no-cache-dir --upgrade pip && \
    pip install --no-cache-dir -r requirements.txt

# Copy Application Source
COPY . .

# Create logs directory
RUN mkdir -p data/logs

EXPOSE 8000

# Healthcheck
HEALTHCHECK --interval=30s --timeout=10s --start-period=5s --retries=3 \
  CMD curl -f http://localhost:8000/ || exit 1

CMD ["uvicorn", "main:app", "--host", "0.0.0.0", "--port", "8000"]
